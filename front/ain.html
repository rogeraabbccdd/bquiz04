# 處理表單

在 api.php 寫入處理管理登入的程式碼
直接複製會員登入來修改比較省時

case "ain":
    // 是否有這個帳號
    $row = All("select * from admin where acc = '".$_POST["acc"]."' and pw = '".$_POST["pw"]."'")[0];
    $num = count($row);
    if($num > 0) 
    {
        // 設定session id為管理者id
        $_SESSION["id"] = $row["id"];
        // 權限
        $_SESSION["permit"] = unserialize($row["permit"]);
        // 導向管理頁
        lo("admin.php");
    }
    else echo "<script>alert('帳號或密碼錯誤'); window.location='index.php?do=ain';</script>";
    break;
